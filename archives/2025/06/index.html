рҹ”§ кі кёү нҷңмҡ© кё°лІ•

BSRR л Ҳм§ҖмҠӨн„° нҷңмҡ©

BSRR(Bit Set/Reset Register)мқ„ мӮ¬мҡ©н•ҳл©ҙ мӣҗмһҗм Ғ м—°мӮ°мқҙ к°ҖлҠҘн•ҳлӢӨ:

// кё°мЎҙ л°©мӢқ (Read-Modify-Write, 3лӢЁкі„ м—°мӮ°)
GPIOA->ODR |= (1U << 5);   // 1. мқҪкё° вҶ’ 2. мҲҳм • вҶ’ 3. м“°кё°
GPIOA->ODR &= ~(1U << 5);  // 1. мқҪкё° вҶ’ 2. мҲҳм • вҶ’ 3. м“°кё°

// BSRR л°©мӢқ (мӣҗмһҗм Ғ м—°мӮ°, 1лӢЁкі„)
GPIOA->BSRR = (1U << 5);       // Set PA5 (м§Ғм ‘ м“°кё°)
GPIOA->BSRR = (1U << (5+16));  // Reset PA5 (м§Ғм ‘ м“°кё°)

BSRR л Ҳм§ҖмҠӨн„° кө¬мЎ°:

Bit 31-16: Reset bits (1 м“°л©ҙ н•ҙлӢ№ н•Җ Reset)
Bit 15-0:  Set bits   (1 м“°л©ҙ н•ҙлӢ№ н•Җ Set)

мҳҲмӢң: BSRR = 0x00100020
- Bit 20 (16+4): GPIOA н•Җ 4лҘј Reset
- Bit 5: GPIOA н•Җ 5лҘј Set

кө¬мЎ°мІҙлҘј мӮ¬мҡ©н•ҳл©ҙ лӢӨмӨ‘ н•Җ м ңм–ҙк°Җ нҡЁмңЁм ҒмқҙлӢӨ:

GPIOA->ODR |= (1U << 5) | (1U << 6) | (1U << 7);  // PA5, PA6, PA7 лҸҷмӢң ON

// BSRRмқ„ мӮ¬мҡ©н•ң лӢӨмӨ‘ н•Җ м ңм–ҙ (лҚ” нҡЁмңЁм Ғ)

GPIOA->BSRR = (1U << 5) | (1U << 6) | (1U << (7+16));  // PA5,6 Set, PA7 Reset

// н•Җлі„ л№„нҠё мң„м№ҳ кі„мӮ° л§ӨнҒ¬лЎң
#define GPIO_MODER_PIN(pin) ((pin) * 2)
#define GPIO_MODER_MASK(pin) (0x3U << GPIO_MODER_PIN(pin))

// к°ңм„ лҗң н•Җ м„Өм • н•ЁмҲҳ
void GPIO_SetMode(GPIO_TypeDef *GPIOx, uint32_t pin, uint32_t mode)
{
    uint32_t pos = GPIO_MODER_PIN(pin);
    GPIOx->MODER &= ~(0x3U << pos);     // кё°мЎҙ к°’ нҒҙлҰ¬м–ҙ
    GPIOx->MODER |= (mode << pos);      // мғҲ к°’ м„Өм •
}

// н•Җ мқҪкё° н•ЁмҲҳ
uint32_t GPIO_ReadPin(GPIO_TypeDef *GPIOx, uint32_t pin)
{
    return ((GPIOx->IDR & (1U << pin)) ? 1 : 0);
}

// н•Җ м“°кё° н•ЁмҲҳ  
void GPIO_WritePin(GPIO_TypeDef *GPIOx, uint32_t pin, uint32_t value)
{
    if (value) {
        GPIOx->BSRR = (1U << pin);        // Set
    } else {
        GPIOx->BSRR = (1U << (pin + 16)); // Reset
    }
}

рҹ“Ҡ м„ұлҠҘкіј л©”лӘЁлҰ¬ л¶„м„қ

л©”лӘЁлҰ¬ мӮ¬мҡ©лҹү л№„көҗ

кө¬мЎ°мІҙлҘј мӮ¬мҡ©н•ҙлҸ„ мӢӨм ң л©”лӘЁлҰ¬ мӮ¬мҡ©лҹүмқҖ ліҖн•ҳм§Җ м•ҠлҠ”лӢӨ:

// л‘ҳ лӢӨ м»ҙнҢҢмқј нӣ„ лҸҷмқјн•ң м–ҙм…Ҳлё”лҰ¬ мҪ”л“ң мғқм„ұ

; мөңм Ғнҷ” м „ (Debug лӘЁл“ң)
ldr r3, =0x40020014    ; GPIOA ODR мЈјмҶҢ лЎңл“ң
ldr r2, [r3]           ; нҳ„мһ¬ к°’ мқҪкё°
orr r2, r2, #32        ; л№„нҠё 5 м„Өм • (1 << 5)
str r2, [r3]           ; к°’ м“°кё°

; мөңм Ғнҷ” нӣ„ (Release лӘЁл“ң, -O2)
mov r2, #32            ; мҰүмӢңк°’ 32 (1 << 5)
str r2, [r3, #24]      ; GPIOA->BSRRм—җ м§Ғм ‘ м“°кё° (лҚ” нҡЁмңЁм Ғ)

BSRR vs ODR м„ұлҠҘ л№„көҗ

ODR мӮ¬мҡ© (Read-Modify-Write):

GPIOA->ODR |= (1U << 5);  // 3 cycles: Read + Modify + Write

BSRR мӮ¬мҡ© (Direct Write):

GPIOA->BSRR = (1U << 5);  // 1 cycle: Write only

м„ұлҠҘ м°Ёмқҙ:

  • BSRRмқҙ ODRліҙлӢӨ м•Ҫ 3л°° л№ лҘҙлӢӨ
  • мӣҗмһҗм Ғ м—°мӮ°мңјлЎң мқён„°лҹҪнҠё м•Ҳм „м„ұ ліҙмһҘ
  • лӢӨмӨ‘ н•Җ м ңм–ҙ мӢң лҚ”мҡұ нҡЁмңЁм Ғ

рҹӣЎпёҸ мқён„°лҹҪнҠё м•Ҳм „м„ұ

л¬ём ң мғҒнҷ©

// мқён„°лҹҪнҠёк°Җ л°ңмғқн•  мҲҳ мһҲлҠ” мғҒнҷ©
GPIOA->ODR |= (1U << 5);   // 3лӢЁкі„ м—°мӮ° мӨ‘к°„м—җ мқён„°лҹҪнҠё л°ңмғқ к°ҖлҠҘ

// л§Ңм•Ҫ мқён„°лҹҪнҠём—җм„ң к°ҷмқҖ л Ҳм§ҖмҠӨн„°лҘј мҲҳм •н•ңлӢӨл©ҙ?
void interrupt_handler() {

  2. лҚ°мқҙн„° мҶҗмӢӨ: мӨ‘к°„м—җ лҒјм–ҙл“  мҲҳм •мңјлЎң мқён•ң к°’ мҶҗмӢӨ
  3. мҳҲмёЎ л¶Ҳк°ҖлҠҘн•ң лҸҷмһ‘: нғҖмқҙл°Қм—җ л”°лқј кІ°кіјк°Җ лӢ¬лқјм§җ

  3. нҡЁмңЁм„ұ: лҚ” л№ лҘё мӢӨн–ү мҶҚлҸ„

   }

SW Stackкіј лӘЁл“Ҳнҷ”

мқҙлІҲ нҸ¬мҠӨнҠём—җм„ңлҠ” ARM л ҲнҸ¬м§ҖнҶ лҰ¬м—җм„ң мӢӨм ңлЎң лӢӨлЈ¬ SW Stack к°ңл…җкіј нҸҙлҚ” л¶„лҰ¬лҘј нҶөн•ң лӘЁл“Ҳнҷ” л°©лІ•мқ„ м•Ңм•„ліёлӢӨ.

рҹ“Ғ нҸҙлҚ” л¶„лҰ¬мқҳ н•„мҡ”м„ұ

GPIO кҙҖл Ё кё°лҠҘ л¶„лҰ¬

кё°мЎҙм—җ main.cм—җ лӘЁл“  мҪ”л“ңк°Җ м§‘мӨ‘лҗҳм–ҙ мһҲлҚҳ кІғмқ„ GPIO кҙҖл Ёлҗң кё°лҠҘл“Өмқ„ л”°лЎң л¶„лҰ¬н•ңлӢӨ.

л¶„лҰ¬ м „ л¬ём ңм җ:

  • лӘЁл“  мҪ”л“ңк°Җ main.cм—җ м§‘мӨ‘
  • кё°лҠҘлі„ кө¬л¶„мқҙ м–ҙл ӨмӣҖ
  • мһ¬мӮ¬мҡ©м„ұ лӮ®мқҢ
  • мң м§ҖліҙмҲҳ м–ҙл ӨмӣҖ

л¶„лҰ¬ нӣ„ мһҘм җ:

  • кё°лҠҘлі„ лӘ…нҷ•н•ң кө¬л¶„
  • мҪ”л“ң мһ¬мӮ¬мҡ© к°ҖлҠҘ
  • мң м§ҖліҙмҲҳ мҡ©мқҙ
  • нҳ‘м—… нҡЁмңЁм„ұ н–ҘмғҒ

рҹҸ—пёҸ SW Stack к°ңл…җ

SWмқҳ нқҗлҰ„кіј кі„мёө кө¬мЎ°

SWмқҳ нқҗлҰ„мқҖ мң„м—җм„ң м•„лһҳлЎң м§„н–үлҗңлӢӨ. SWк°Җ HWлҘј м ңм–ҙн•ҳлҠ” кі„мёөм Ғ кө¬мЎ°лӢӨ.

к°Ғ кі„мёөмқҳ м—ӯн• 

main кі„мёө:

  • system мҙҲкё°нҷ”
  • application мӢӨн–ү кё°лҠҘ мҷём—җлҠ” кі л Өн•ҳм§Җ м•ҠлҠ”лӢӨ

Application кі„мёө:

  • мӢӨм ң м• н”ҢлҰ¬мјҖмқҙм…ҳ лЎңм§Ғ кө¬нҳ„
  • л№„мҰҲлӢҲмҠӨ мҡ”кө¬мӮ¬н•ӯ мІҳлҰ¬

Driver кі„мёө:

  • н•ҳл“ңмӣЁм–ҙ м¶”мғҒнҷ”
  • мһ¬мӮ¬мҡ© к°ҖлҠҘн•ң кё°лҠҘ м ңкіө

Interface кі„мёө:

  • н•ҳл“ңмӣЁм–ҙ м§Ғм ‘ м ңм–ҙ
  • л Ҳм§ҖмҠӨн„° мҲҳмӨҖ м ‘к·ј

рҹҡ« мһҳлӘ»лҗң м ‘к·ј л°©мӢқ

Applicationм—җм„ң Interface м§Ғм ‘ м ‘к·ј л¬ём ң

Applicationм—җм„ң л°”лЎң GPIO(Interface)лЎң м ‘к·јн•ҳлҠ” кІғмқҖ м§Җм–‘н•ҙм•ј н•ңлӢӨ.

л¬ём ңм җ:

  • GPIOлҠ” н•ҳл“ңмӣЁм–ҙм—җ к°Җк№Ңмҡҙ л¶Җл¶„мқҙкё° л•Ңл¬ём—җ, Applicationм—җм„ң м§Ғм ‘ м ‘к·јн•ҳлҠ” кІғмқҖ мўӢм§Җ м•ҠлӢӨ
  • лӢЁм§Җ Driverл§Ң мҲҳм •н•ҳл©ҙ лҗҳлҠ” кІғмқҙ Interfaceк№Ңм§Җ мҲҳм •н•ҙм•ј лҗҳм–ҙ лІҲкұ°лЎңмӣҢм§„лӢӨ

л„ӨнҠёмӣҢнҒ¬ мҠӨнғқ мҳҲмӢң

мқён„°л„·мқҳ вҖҳsoftware stackвҖҷм—җм„ңлҸ„ лҸҷмқјн•ң мӣҗм№ҷмқҙ м Ғмҡ©лҗңлӢӨ.

Applicationм—җм„ң л°”лЎң MACмқҙлӮҳ PHYм—җ м ‘к·јн•ҳм§Җ м•ҠлҠ”лӢӨ.

рҹ’Ў мҳ¬л°”лҘё м„Өкі„ мӣҗм№ҷ

мқём ‘н•ң LayerлҘј нҶөн•ң нҶөмӢ 

н•өмӢ¬ мӣҗм№ҷ: Applicationм—җм„ң мқём ‘н•ң LayerлҠ” DriverлӢӨ.

мқём ‘н•ң LayerлҘј нҶөн•ҙ л©”мӢңм§ҖлҘј мҶЎмҲҳмӢ н•ҙм•ј н•ңлӢӨ

мқҙл ҮкІҢ л§Ңл“Өм–ҙм•ј мһ¬мӮ¬мҡ©м„ұмқҙ мўӢлӢӨ.

рҹ”§ мӢӨм ң м Ғмҡ©: GPIO лІ„нҠј м¶”к°Җ

кё°мЎҙ л¬ём ңмғҒнҷ©

кё°мЎҙмқҳ LED Shifter кё°лҠҘм—җм„ң BTN кё°лҠҘмқ„ м¶”к°Җн•ҳм—¬ BTNмқ„ лҲ„лҘҙл©ҙ LEDк°Җ л°ҳлҢҖ л°©н–ҘмңјлЎң SHIFT н•ҳлҸ„лЎқ н•ҳл Өкі  н–ҲлӢӨ.

н•ҳм§Җл§Ң кё°мЎҙм—җ кө¬нҳ„н–ҲлҚҳ кІғмқҖ лІ„нҠјмқ„ лҲ„лҘҙкі  мһҲмқ„ л•Ң LEDк°Җ л©Ҳм¶”лҠ” л¬ём ңк°Җ мһҲм—ҲлӢӨ.

BTN кё°лҠҘ м¶”к°ҖлҘј мң„н•ң SW Stack

ap_main кі„мёө м¶”к°Җ

ap_mainмқ„ л§Ңл“Өм–ҙм„ң SW Stack LayerлҘј н•ҳлӮҳ лҚ” м¶”к°Җн•ңлӢӨ.

main.cмқҳ кё°лҠҘ:

  • system мҙҲкё°нҷ”
  • application мӢӨн–ү кё°лҠҘ мҷём—җлҠ” кі л Өн•ҳм§Җ м•ҠлҠ”лӢӨ

рҹ”ҳ лІ„нҠј н•ҳл“ңмӣЁм–ҙ мқҙн•ҙ

BTN(Push Switch, Tact Switch) л¶„м„қ

SchematicмңјлЎң мҡ°лҰ¬к°Җ к°Җм§„ BTNмқҙ HWм ҒмңјлЎң м–ҙл–Ө нҳ•нғңмқём§Җ м•Ң мҲҳ мһҲлӢӨ.

BTNмқҙ Pullup лҗҳм–ҙмһҲлӢӨ:

  • BTNмқҙ лҲҢлҰ¬м§Җ м•ҠмқҖ мғҒнғң: 3.3V (HIGH)
  • BTNмқҙ лҲҢлҰ° мғҒнғң: 0V (LOW)

Chattering(Bounce) нҳ„мғҒ

мқҙмғҒм ҒмңјлЎң 0 лҳҗлҠ” 1лЎң ліҖнҷ”н•ҳл©ҙ мҷ„лІҪн•ҳм§Җл§Ң л…ёмқҙмҰҲк°Җ к»ҙм„ң Chattering(Bounce) нҳ„мғҒмқҙ л°ңмғқн•ңлӢӨ.

мқҙлҹ¬н•ң Noise лҳҗн•ң кі л Өн•ҳл©ҙм„ң мҪ”л“ңлҘј м§ңм•јн•ңлӢӨ. (debounce мҪ”л“ңлҘј мһ‘м„ұн•ҙм•јн•ңлӢӨ)

рҹ”Қ лІ„нҠј мғҒнғң к°җм§Җ кө¬нҳ„

лІ„нҠј мғҒнғң м •мқҳ

BTNмқҳ мғҒнғң:

  1. Button Normal мғҒнғң: RELEASED(1) мғҒнғң
  2. Button Push: PUSHED(0) мғҒнғң

GPIO READ н•ЁмҲҳ кө¬нҳ„

лІ„нҠјмқҳ мғҒнғңлҘј л°ӣкі  мӢ¶лӢӨ. Button_GetState(): лІ„нҠјмқҳ мғҒнғңлҘј л°ӣм•„мҳӨлҠ” н•ЁмҲҳ

uint32_t GPIO_ReadPin(GPIO_TypeDef *GPIOx, uint32_t pinNum){
    return ((GPIOx -> IDR & (1U << pinNum))? 1:0); //0мқҙл©ҙ 0мқҙлӮҳк°Җкі , 0мқҙ м•„лӢҢ к°’мқҙл©ҙ 1мқҙлӮҳк°„лӢӨ
}

Button Get State н•ЁмҲҳ

button_state_t Button_GetState(){
 static uint32_t prevState = RELEASED; //м „мӣҗмқ„ мІҳмқҢм—җ л„Јмңјл©ҙ мҙҲкё°к°’мқҙ HIGH
 uint32_t curState;
 curState= GPIO_ReadPin(GPIOC, 13);

 //мІҳмқҢ лҲ„лҘё кІҪмҡ°
 if ((prevState == RELEASED) && (curState == PUSHED)){
  prevState = PUSHED;
  return ACT_PUSHED;
 } else if ((prevState == PUSHED) &&(curState == RELEASE)){
  prevState = RELEASED;
  return ACT_RELEASED;
 }
 return NO_ACT;
}

static ліҖмҲҳ мӮ¬мҡ© мқҙмң :

  • ScopeлҘј лІ—м–ҙлӮҳлҸ„ л©”лӘЁлҰ¬ кіөк°„мқҙ л°ҳнҷҳлҗҳм§Җ м•ҠлҠ”лӢӨ
  • л©”лӘЁлҰ¬ мһ¬н• лӢ№мқ„ м•Ҳн•ҳлӢҲ мқҙм „ к°’мқҙ кі„мҶҚ мң м§ҖлҗңлӢӨ

рҹ’» Cм–ём–ҙмқҳ н•ңкі„

Edge Detectмқҳ н•ңкі„

  • Cм–ём–ҙ мҪ”л“ңл§ҢмңјлЎңлҠ” edge detect н•ҳлҠ”кІҢ м—ҶлӢӨ
  • Cм–ём–ҙлЎң CLK edge лҸҷкё°нҷ”лҘј н•  мҲҳ м—ҶлӢӨ

мқҙлҹ¬н•ң н•ңкі„ л•Ңл¬ём—җ мҶҢн”„нҠёмӣЁм–ҙм ҒмңјлЎң мғҒнғң ліҖнҷ”лҘј к°җм§Җн•ҙм•ј н•ңлӢӨ.

рҹ“Ӣ м •лҰ¬

мқҙлІҲ нҸ¬мҠӨнҠём—җм„ңлҠ” SW Stackкіј лӘЁл“Ҳнҷ”мқҳ мӨ‘мҡ”м„ұмқ„ лӢӨлӨҳлӢӨ:
